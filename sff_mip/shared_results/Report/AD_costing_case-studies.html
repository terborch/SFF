Economic considerations¶

The operating cost is a significant part of the total cost of the biogas cleaning process. It is therfore useful to annualize the investment cost in order to arrive to the total anual cost or TOTEX to be able to accuratly compare each scenario.

# functions
def write_EX(inv, op, S, i, unit):
    if unit == 'energy':
        # in EUR/kW-LHV/year
        TOTEX_q[i] = Tau*inv + op
        # in EUR/kW-LHV
        CAPEX_q[i] = inv
        # in EUR/kW-LHV/year
        OPEX_q[i] = op
        # in kW-LHV
        Size_q[i] = S
    else:
        # in EUR/m^3/h/year
        TOTEX_v[i] = Tau*inv + op
        # in EUR/m^3/h
        CAPEX_v[i] = inv
        # in EUR/m^3/h/year
        OPEX_v[i] = op
        # in m^3/h
        Size_v[i] = S
def print_results(inv,  op, S, unit):
    if unit == 'energy':
        print('Function of energy flow: Cost per kW-LHV of biogas produced:')
    else:
        print('Function of volume flow: Cost per m^3/h of biogas produced:')
    
    print('specific investment cost: ', ['{:.2f}'.format(i) for i in inv])
    print('specific operating cost: ', ['{:.2f}'.format(i) for i in op])
    print('size: ', ['{:.2f}'.format(i) for i in S])

# Switch functions for the different graphical representations
def switch_kind(kind):
    switcher={
        'TOTEX': 'Total Annual Costs',
        'CAPEX': 'Investment Costs',
        'OPEX': 'Operating Costs',
    }
    return switcher.get(kind, "Invalid kind")

def switch_unit(unit):
    switcher={
        'energy': ['Energy Flow', 'kW-LHV'],
        'volume': ['Volume Flow', ' $m^3$/ h '],
    }
    return switcher.get(unit, "Invalid unit")

def plt_scatter_data(Size, Cost, Accuracy, paper, tight, kind, unit, fit):
    fig, ax = plt.subplots()
    # scatter plot of paper data
    for i in np.arange(len(Size)):
        ax.scatter(Size[i], Cost[i], c = 'green', label = paper[i][0], marker = paper[i][1])
    
    # plot the given scaling law
    if fit != 0:
        plt_fit_law(fit, Size, Cost, unit)
    
    # reduce size to the scale of focus
    if tight:
        S = [i for s in Size for i in s]
        S.sort()
        plt.xlim(0, S[-1]*1.1)
        
        C = [i for s in Cost for i in s]
        C.sort()
        plt.ylim(0, C[-1]*1.25)
    
    # create a plot title and axis names according to kind and unit
    u = switch_unit(unit)
    k = switch_kind(kind)
    
    Title = ' '.join([k, 'vs', u[0]])
    Xlabel = '{} [ {} ]'.format(u[0], u[1])
    if kind == 'CAPEX': 
        Ylabel = '{} [ EUR / {} ]'.format(kind, u[1])
    else:
        Ylabel = '{} [ EUR / {} / year ]'.format(kind, u[1])
    
    plt.title(Title)
    plt.xlabel(Xlabel)
    plt.ylabel(Ylabel)
    ax.yaxis.grid(True, linestyle='-', which='major', color='lightgrey',alpha=0.5)
    
    plt.legend(loc = 'upper right')
    
    file_name = '{}_vs_size_{}_AD_case.pdf'.format(kind, unit)
    if save: 
        plt.savefig(file_name)
        print('figure saved as', file_name)
