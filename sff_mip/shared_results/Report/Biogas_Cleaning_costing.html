# functions
def write_EX(inv, op, S, i, unit):
    if unit == 'energy':
        # in EUR/kW-LHV/year
        TOTEX_q[i] = Tau*inv + op
        # in EUR/kW-LHV
        CAPEX_q[i] = inv
        # in EUR/kW-LHV/year
        OPEX_q[i] = op
        # in kW-LHV
        Size_q[i] = S
    else:
        # in EUR/m^3/h/year
        TOTEX_v[i] = Tau*inv + op
        # in EUR/m^3/h
        CAPEX_v[i] = inv
        # in EUR/m^3/h/year
        OPEX_v[i] = op
        # in m^3/h
        Size_v[i] = S
def print_results(inv,  op, S, unit):
    if unit == 'energy':
        print('Function of energy flow in kW-LHV of biogas:')
    else:
        print('Function of volume flow in m^3/h of biogas:')
    
    print('investment cost: ', ['{:.2f}'.format(i) for i in inv])
    print('operating cost: ', ['{:.2f}'.format(i) for i in op])
    print('size: ', ['{:.2f}'.format(i) for i in S])

2. N. de Arespacochaga 2014¶

Title : Biogas deep clean-up based on adsorption technologies for Solid Oxide Fuel Cell applications

Authors : N. de Arespacochaga, C. Valderrama, C. Mesa, L. Bouchy, J.L. Cortina

Year of publication : 2014

Region of study : Spain

Relevant summary : Installation of a small scale biogas cleaning unit for SOFC applications, downstream of an anaerobic digester in a WWTP in Spain. The set-up is (i) a regenerable iron-based adsorbent unit to remove H2S, (ii) a biogas drying unit to remove moisture and (iii) an activated carbon unit to remove the remaining trace components (siloxanes, linear and aromatic hydrocarbons). H2S is removed to below 0.5ppm (average 0.1ppm) and Siloxanes, Linear HC and aromatic HC to 1.4-0 ppb. The experiment was conducted for 10m^3 biogas/h for 3600 hours and and economic evaluation, using a scaling factor, predicted the cost for a biogas cleaning process with a though-ut of 190m^3/h. Results for three scenarios are presented: (A) no pre-treatment in the AD. (B) some pre-treatment using bio-trickling. (C) the pre-treatment is optimized to reach 200ppm of solfure in the biogas. Costs for all scenarios are presented at both scales.

def switch_kind(kind):
    switcher={
        'TOTEX': 'Total Annual Costs',
        'CAPEX': 'Investment Costs',
        'OPEX': 'Operating Costs',
    }
    return switcher.get(kind, "Invalid kind")

    switcher={
        'energy': ['Energy Flow', 'kW-LHV'],
        'volume': ['Volume Flow', ' $m^3$/ h '],
    }
    return switcher.get(unit, "Invalid unit")

def plt_scatter_data(Size, Cost, Accuracy, paper, tight, kind, unit, fit):
    fig, ax = plt.subplots()
    # scatter plot of paper data
    for i in np.arange(len(Size)):
        ax.scatter(Size[i], Cost[i], c = Accuracy[i], label = paper[i][0], marker = paper[i][1])
    
    # plot the given scaling law
    if fit != 0:
        plt_fit_law(fit, Size, Cost, unit)
    
    # reduce size to the scale of focus
    if tight:
        S = [i for s in Size for i in s]
        S.sort()
        plt.xlim(0, S[-3]*1.1)
        
        C = [i for s in Cost for i in s]
        C.sort()
        plt.ylim(0, C[-1]*1.25)
    
    # create a plot title and axis names according to kind and unit
    u = switch_unit(unit)
    k = switch_kind(kind)
    
    Title = ' '.join([k, 'vs', u[0]])
    Xlabel = '{} [ {} ]'.format(u[0], u[1])
    if kind == 'CAPEX': 
        Ylabel = '{} [ EUR / {} ]'.format(kind, u[1])
    else:
        Ylabel = '{} [ EUR / {} / year ]'.format(kind, u[1])
    
    plt.title(Title)
    plt.xlabel(Xlabel)
    plt.ylabel(Ylabel)
    ax.grid(True, linestyle='-', which='major', color='lightgrey',alpha=0.5)
    
    plt.legend(loc = 'upper center')
    
    file_name = '{}_vs_size_{}.pdf'.format(kind, unit)
    if save: 
        plt.savefig(file_name)
        print('figure saved as', file_name)

def tabulate(cost):
    tab = {}
    for i in range(len(paper) - 1):
        values = cost[i].tolist()
        values = [int(round(j)) for j in values]
        
        if i == 1:
            tab[paper[i][0] + ' small'] = values[:3]
            tab[paper[i][0] + ' large'] = values[3:]
        else:
            tab.update({paper[i][0]: values})
        
    return tab
